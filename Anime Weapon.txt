function AutoFind()
	while _G.autoclick == true do
		repeat
			game:GetService("ReplicatedStorage").System.SystemClick.Click:FireServer()
			wait(0.1)
		until _G.autoclick == false
	end
end

function autoHatch()
	while _G.autoHatch == true do
		repeat
			game:GetService("ReplicatedStorage").Tool.DrawUp.Msg.DrawHero:InvokeServer(7000000 + numbersss,1)
			wait(0.1)
		until _G.autoHatch == false
	end
end

function AutoFarmWorld1()
	while _G.World1Farm == true do
		repeat
			for i,v in pairs (game.Workspace.Enemy["1"]:GetChildren()) do
				if v.EnemyTag.NameLabel.Text == _G.World1Monster then
					local player = game.Players.LocalPlayer 
					repeat
						player.Character.HumanoidRootPart.CFrame = v.HumanoidRootPart.CFrame
						wait(0.1)
					until v.HP.Value == 0 
				end
			end
			wait(0.1)
		until _G.World1Farm == false
	end
end
